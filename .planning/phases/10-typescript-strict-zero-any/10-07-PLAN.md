**Create `eslint.config.js` at workspace root:**

```javascript
import js from '@eslint/js';
import tseslint from 'typescript-eslint';

export default tseslint.config(
  // Ignore vendor and build artifacts
  { ignores: ['packages/pi-*/**', 'packages/native/**', 'dist/**', 'dist-test/**', 'node_modules/**'] },

  // GSD packages: strict typed linting
  {
    files: [
      'packages/gsd-agent-core/src/**/*.ts',
      'packages/gsd-agent-modes/src/**/*.ts',
      'packages/gsd-agent-types/src/**/*.ts',
      'src/**/*.ts',
    ],
    extends: [js.configs.recommended, ...tseslint.configs.recommended],
    languageOptions: {
      parserOptions: {
        project: true,
        tsconfigRootDir: import.meta.dirname,
      },
    },
    rules: {
      '@typescript-eslint/no-explicit-any': 'error',
      '@typescript-eslint/explicit-function-return-type': ['error', {
        allowExpressions: true,
        allowTypedFunctionExpressions: true,
        allowHigherOrderFunctions: true,
      }],
      '@typescript-eslint/switch-exhaustiveness-check': 'error',
      '@typescript-eslint/ban-ts-comment': [
        'error',
        {
          'ts-ignore': 'allow-with-description',
          'ts-expect-error': 'allow-with-description',
          minimumDescriptionLength: 15,
        },
      ],
    },
  },
);
```

**Step 2 — Pi-* modification rejection (D-03):**
```yaml
      - name: Reject pi-* modifications
        if: github.event_name == 'pull_request'
        env:
          PR_BASE_SHA: ${{ github.event.pull_request.base.sha }}
        run: |
          CHANGED=$(git diff --name-only "$PR_BASE_SHA"...HEAD | grep '^packages/pi-' || true)
          if [ -n "$CHANGED" ]; then
            echo "::error::PR modifies read-only pi-* packages. Submit changes upstream."
            echo "$CHANGED"
            exit 1
          fi
```

The `if: github.event_name == 'pull_request'` guard ensures this only runs on PRs, not on push events (which may legitimately include pi-* changes during vendor updates like Phase 07).
