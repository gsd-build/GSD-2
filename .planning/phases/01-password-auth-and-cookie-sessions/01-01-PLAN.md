    **web-auth-storage.ts** (`src/web/web-auth-storage.ts`):

    Addresses review concern #4: Auth data stored in `~/.gsd/web-auth.json`, NOT `web-preferences.json`. The existing GET /api/preferences endpoint returns raw JSON from web-preferences.json — storing the password hash there would leak it to the client.

    ```typescript
    import { readFile, writeFile, mkdir } from 'node:fs/promises'
    import { join } from 'node:path'
    import { hashPassword, rotateSessionSecret } from './web-session-auth.js'
    import { appRoot } from '../app-paths.js'

    const AUTH_FILE = 'web-auth.json'

    function authFilePath(gsdDir?: string): string {
      return join(gsdDir ?? appRoot, AUTH_FILE)
    }

    export async function setPassword(password: string, gsdDir?: string): Promise<void> {
      const dir = gsdDir ?? appRoot
      const filePath = authFilePath(dir)
      const passwordHash = await hashPassword(password)

      // Read-modify-write to preserve any other keys
      let data: Record<string, unknown> = {}
      try {
        const raw = await readFile(filePath, 'utf-8')
        data = JSON.parse(raw)
      } catch {
        // File doesn't exist yet
      }

      data.passwordHash = passwordHash
      await mkdir(dir, { recursive: true })
      await writeFile(filePath, JSON.stringify(data, null, 2), { mode: 0o600 })

      // Rotate session secret to invalidate all existing sessions (AUTH-07)
      await rotateSessionSecret(dir)
    }

    export async function getPasswordHash(gsdDir?: string): Promise<string | null> {
      try {
        const raw = await readFile(authFilePath(gsdDir), 'utf-8')
        const data = JSON.parse(raw)
        if (typeof data.passwordHash === 'string' && data.passwordHash.includes(':')) {
          return data.passwordHash
        }
        return null
      } catch {
        return null
      }
    }
    ```

    Addresses review concern (MEDIUM) about underspecified rate limiter API:
    - `checkRateLimit` only counts calls (it does NOT distinguish success/failure). The login route calls it on every attempt.
    - The returned `resetAt` timestamp tells the client when the window expires (for D-10 countdown).
    - `_testResetRateLimits()` is exported for test cleanup.

    ```typescript
    const loginAttempts = new Map<string, { count: number; resetAt: number }>()

    export function checkRateLimit(ip: string): { allowed: boolean; resetAt: number } {
      const now = Date.now()
      const record = loginAttempts.get(ip)

      if (!record || now > record.resetAt) {
        loginAttempts.set(ip, { count: 1, resetAt: now + 60_000 })
        return { allowed: true, resetAt: now + 60_000 }
      }

      if (record.count >= 5) {
        return { allowed: false, resetAt: record.resetAt }
      }

      record.count++
      return { allowed: true, resetAt: record.resetAt }
    }

    /** @internal test-only — clears all rate limit state */
    export function _testResetRateLimits(): void {
      loginAttempts.clear()
    }
    ```
